function calculateTax() { var total = parseInt(document.getElementById("total").value); var taxRate = parseInt(document.getElementById("taxRate").value); var option = document.getElementById("option").value; if (isNaN(total) || isNaN(taxRate)) { alert("Please enter valid numbers."); return; } var tax = (total / 100) * taxRate; if (option === "printTax") { console.log("The tax charged in profit Rs.", tax); } else if (option === "showValue") { document.getElementById("result").innerHTML = "The tax charged in profit Rs. " + tax; } }
